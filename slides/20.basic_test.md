### Basic Test (cont.)

```javascript
var expect = require('chai').expect;
var validator = require('../../app/validators/template');
```

```javascript
describe("Template validator", function () {

  describe("when add new template", function () {

    it('should raise error when miss name', function () {
      try {
        validator.isValidTemplateWhenAddNew({});
      } catch (error) {
        expect(error).not.be.null;
        expect(error.message).to.match(/Name is required/g);
      }
    });

    it('should return true when param is valid', function () {
      var param = { name: "test" };
      expect(validator.isValidTemplateWhenAddNew(param)).to.be.true;
    });

  });

});
```
